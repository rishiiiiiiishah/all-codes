<script>
// JavaScript conditional statement to find the sign of the product of three numbers
let num1 = 2;
let num2 = 3;
let num3 = 5;

let product = num1 * num2 * num3;

if (product > 0) {
    alert("The product is positive");
} else if (product < 0) {
    alert("The product is negative");
} else {
    alert("The product is zero");
}
</script>
